local Loops = {}
local LoopInfo = {}
local SortedLoops = {}
local NeedSort = false

local function SortLoops()
    table.clear(SortedLoops)
    for name in pairs(Loops) do
        table.insert(SortedLoops, name)
    end

    table.sort(SortedLoops, function(a, b)
        return (LoopInfo[a].priority or 1) > (LoopInfo[b].priority or 1)
    end)

    NeedSort = false
end

local function AddLoop(name, func, delay, priority)
    Loops[name] = func
    LoopInfo[name] = {
        delay = delay or 0,
        last = 0,
        priority = priority or 1,
        enabled = true
    }
    NeedSort = true
end

local function RemoveLoop(name)
    Loops[name] = nil
    LoopInfo[name] = nil
    NeedSort = true
end

local function ToggleLoop(name, state)
    if LoopInfo[name] then
        LoopInfo[name].enabled = state
    end
end

local function PauseAllLoops(state)
    for _, info in pairs(LoopInfo) do
        info.enabled = not state
    end
end

RunService.Heartbeat:Connect(function()
    if NeedSort then
        SortLoops()
    end

    local now = tick()

    for _, name in ipairs(SortedLoops) do
        local func = Loops[name]
        local info = LoopInfo[name]

        if func and info and info.enabled then
            if now - info.last >= info.delay then
                info.last = now

                local success, err = pcall(func)
                if not success then
                    warn("Loop Error ["..name.."]:", err)
                end
            end
        end
    end
end)

AimSystem = {}

function AimSystem.RotateTo(targetCFrame, speed)
    local myHRP = GetRootPart()
    if not myHRP or not targetCFrame then return end

    speed = speed or 1

    local currentCFrame = myHRP.CFrame
    local goalCFrame = CFrame.new(currentCFrame.Position, targetCFrame.Position)

    myHRP.CFrame = currentCFrame:Lerp(goalCFrame, math.clamp(speed, 0, 1))
end

function AimSystem.GetTargetCFrame(target)
    if not target then return nil end

    -- Hỗ trợ cả Player và Model (AI/Killer)
    if target:IsA("Player") then
        local char = target.Character
        if not char then return nil end

        local hrp = char:FindFirstChild("HumanoidRootPart")
        if hrp then
            return hrp.CFrame
        end

    elseif target:IsA("Model") then
        local hrp = target:FindFirstChild("HumanoidRootPart")
        if hrp then
            return hrp.CFrame
        end
    end

    return nil
end

local GuiClick = {}

GuiClick._cache = {}
GuiClick._initialized = false

function GuiClick:Init()
    if self._initialized then return end
    self._initialized = true

    local function scan()
        local gui = Services.PlayerGui
        if not gui then return end

        local mainUI = gui:FindFirstChild("MainUI")
        local container = mainUI and mainUI:FindFirstChild("AbilityContainer")
        if not container then return end

        for _, btn in ipairs(container:GetChildren()) do
            if btn:IsA("ImageButton") then
                if not self._cache[btn.Name] then
                    local info = {
                        button = btn,
                        connections = {},
                        remote = nil
                    }

                    local success, conns = pcall(getconnections, btn.MouseButton1Click)
                    if success and conns then
                        info.connections = conns

                        for _, conn in ipairs(conns) do
                            local f = conn.Function
                            if f and islclosure(f) then
                                for _, v in pairs(getupvalues(f)) do
                                    if typeof(v) == "Instance" and v:IsA("RemoteEvent") then
                                        info.remote = v
                                        break
                                    end
                                end
                            end
                        end
                    end

                    self._cache[btn.Name] = info
                end
            end
        end
    end

    scan()

    LocalPlayer.CharacterAdded:Connect(function()
        task.wait(0.5)
        self._cache = {}
        scan()
    end)
end

function GuiClick:Use(skillName)
    local data = self._cache[skillName]
    if not data then return false end

    if data.remote then
        pcall(function()
            data.remote:FireServer(true)
            task.delay(0.05, function()
                data.remote:FireServer(false)
            end)
        end)
        return true
    end

    for _, conn in ipairs(data.connections or {}) do
        pcall(function()
            conn:Fire()
        end)
    end

    if data.button then
        pcall(function()
            data.button:Activate()
        end)
    end

    return true
end

AnimationService = {}
AnimationService.__index = AnimationService

AnimationService.Character = Character
AnimationService.Humanoid = Humanoid
AnimationService.Animator = Humanoid and Humanoid:FindFirstChildOfClass("Animator")

AnimationService.Tracks = {}
AnimationService.Animations = {}
AnimationService.Enabled = {}
AnimationService.Options = {}

AnimationService.PriorityLock = false
AnimationService.BlockLowerPriorityOnly = true

local function setupAnimator()
    if not AnimationService.Humanoid then return end

    AnimationService.Animator =
        AnimationService.Humanoid:FindFirstChildOfClass("Animator")
        or Instance.new("Animator", AnimationService.Humanoid)

    for _, track in pairs(AnimationService.Tracks) do
        pcall(function()
            track:Stop(0.1)
            track:Destroy()
        end)
    end

    table.clear(AnimationService.Tracks)
end

function AnimationService:_blockOtherAnimations(currentTrack)
    if not self.PriorityLock or not self.Animator then return end

    for _, track in ipairs(self.Animator:GetPlayingAnimationTracks()) do
        if track ~= currentTrack then
            local isOurTrack = false

            for _, t in pairs(self.Tracks) do
                if t == track then
                    isOurTrack = true
                    break
                end
            end

            if not isOurTrack then
                if self.BlockLowerPriorityOnly then
                    if track.Priority < currentTrack.Priority then
                        pcall(function() track:Stop(0.1) end)
                    end
                else
                    pcall(function() track:Stop(0.1) end)
                end
            end
        end
    end
end

function AnimationService:Init()
    setupAnimator()

    LocalPlayer.CharacterAdded:Connect(function(char)
        task.wait(0.15)

        self.Character = char
        self.Humanoid = char:WaitForChild("Humanoid")

        setupAnimator()

        for name, enabled in pairs(self.Enabled) do
            if enabled then
                self:_playInternal(name)
            end
        end
    end)
end

function AnimationService:Register(name, animationId, options)
    self.Animations[name] = tostring(animationId)
    self.Options[name] = options or {}
end

function AnimationService:_load(name)
    if self.Tracks[name] then
        return self.Tracks[name]
    end

    local animId = self.Animations[name]
    if not animId or not self.Animator then return nil end

    local anim = Instance.new("Animation")
    anim.AnimationId = "rbxassetid://" .. animId

    local track = self.Animator:LoadAnimation(anim)
    self.Tracks[name] = track

    return track
end

function AnimationService:_playInternal(name)
    local track = self:_load(name)
    if not track then return end

    local opt = self.Options[name] or {}

    if opt.Priority then
        track.Priority = opt.Priority
    else
        track.Priority = Enum.AnimationPriority.Action
    end

    track:Play(opt.FadeTime or 0.15)
    if opt.Speed then
        track:AdjustSpeed(opt.Speed)
    end

    self:_blockOtherAnimations(track)

    track.Stopped:Connect(function()
        if self.Enabled[name] then
            track:Play(0.1)
            self:_blockOtherAnimations(track)
        end
    end)

    return track
end

function AnimationService:Enable(name)
    if not self.Animations[name] then return end
    self.Enabled[name] = true
    self:_playInternal(name)
end

function AnimationService:Disable(name)
    self.Enabled[name] = false

    local track = self.Tracks[name]
    if track then
        pcall(function()
            track:Stop(0.15)
            track:Destroy()
        end)
    end

    self.Tracks[name] = nil
end

function AnimationService:IsEnabled(name)
    return self.Enabled[name] == true
end

function AnimationService:StopAll()
    for name in pairs(self.Enabled) do
        self:Disable(name)
    end
end

function AnimationService:SetPriorityLock(state)
    self.PriorityLock = state == true
end

local ESPManager = {
    ActiveTypes = {},
    Objects = {},
    Filters = {},
    Colors = {},
    Watchers = {},
    ShowHP = {},

    TypeConfig = {},
    KnownModels = {},

    _pendingCreate = {},
}

local Shared = getgenv().__HutaoShared or {}
getgenv().__HutaoShared = Shared
Shared.ESP = ESPManager

local function getHRP()
    return GetRootPart()
end

local function getPrimaryPart(model)
    return model
        and (model.PrimaryPart
        or model:FindFirstChild("HumanoidRootPart")
        or model:FindFirstChildWhichIsA("BasePart"))
end

local function hasHumanoid(model)
    return model and model:FindFirstChildOfClass("Humanoid")
end

local function isAlive(model)
    local hum = hasHumanoid(model)
    return not hum or hum.Health > 0
end

local function disconnect(tbl)
    for _, c in pairs(tbl or {}) do
        if typeof(c) == "RBXScriptConnection" then
            pcall(function()
                c:Disconnect()
            end)
        end
    end
end

function ESPManager:RegisterType(name, color, filterFn, showHP)
    self.Filters[name] = filterFn
    self.Colors[name] = color or Color3.new(1, 1, 1)
    self.ShowHP[name] = showHP or false
    self.ActiveTypes[name] = false

    self.TypeConfig[name] = {
        Text = true,
        Highlight = true,
    }

    self.KnownModels[name] = {}
end

function ESPManager:_CreateImmediate(model, typeName)
    if not model then return end
    if not self.ActiveTypes[typeName] then return end
    if not isAlive(model) then return end
    if self.Objects[model] then return end

    local part = getPrimaryPart(model)
    if not part then return end

    local cfg = self.TypeConfig[typeName]
    if not cfg then return end

    local color = self.Colors[typeName]
    local conns = {}
    local gui, label, hl

    if cfg.Text then
        gui = Instance.new("BillboardGui")
        gui.Size = UDim2.new(0, 240, 0, 40)
        gui.AlwaysOnTop = true
        gui.StudsOffset = Vector3.new(0, 4, 0)
        gui.MaxDistance = 800
        gui.Parent = Services.CoreGui

        label = Instance.new("TextLabel")
        label.Size = UDim2.fromScale(1, 1)
        label.BackgroundTransparency = 1
        label.TextColor3 = color
        label.TextStrokeTransparency = 0.25
        label.Font = Enum.Font.GothamBold
        label.TextSize = 14
        label.Parent = gui

        table.insert(conns, Services.RunService.RenderStepped:Connect(function()
            if not model or not model.Parent then
                ESPManager:Remove(model)
                return
            end
            local p = getPrimaryPart(model)
            if not p then
                ESPManager:Remove(model)
                return
            end
            gui.Adornee = p
        end))
    end

    if cfg.Highlight then
        hl = Instance.new("Highlight")
        hl.Adornee = model
        hl.FillColor = color
        hl.OutlineColor = color
        hl.FillTransparency = 0.65
        hl.DepthMode = Enum.HighlightDepthMode.AlwaysOnTop
        hl.Parent = Services.CoreGui
    end

    table.insert(conns, model.AncestryChanged:Connect(function(_, parent)
        if not parent then
            ESPManager:Remove(model)
            ESPManager.KnownModels[typeName][model] = nil
        end
    end))

    table.insert(conns, model:GetPropertyChangedSignal("Parent"):Connect(function()
        if not model.Parent then
            ESPManager:Remove(model)
        end
    end))

    local hum = hasHumanoid(model)
    if hum then
        table.insert(conns, hum.Died:Connect(function()
            ESPManager:Remove(model)
        end))
    end

    self.Objects[model] = {
        type = typeName,
        gui = gui,
        label = label,
        hl = hl,
        conns = conns,
    }
end

function ESPManager:_ScheduleCreate(model, typeName)
    if not model then return end
    if not self.ActiveTypes[typeName] then return end
    if self._pendingCreate[model] then return end
    if self.Objects[model] then return end

    self._pendingCreate[model] = true

    task.delay(0.05, function()
        self._pendingCreate[model] = nil
        if not self.ActiveTypes[typeName] then return end

        local f = self.Filters[typeName]
        if model and model.Parent and f and f(model) then
            ESPManager:_CreateImmediate(model, typeName)
        end
    end)
end

function ESPManager:Remove(model)
    local d = self.Objects[model]
    if not d then return end

    disconnect(d.conns)
    if d.gui then pcall(function() d.gui:Destroy() end) end
    if d.hl then pcall(function() d.hl:Destroy() end) end

    self.Objects[model] = nil
    self._pendingCreate[model] = nil
end

function ESPManager:StartWatcher(typeName)
    if self.Watchers[typeName] then return end
    local f = self.Filters[typeName]
    local cache = self.KnownModels[typeName]

    for _, obj in ipairs(Services.Workspace:GetDescendants()) do
        if f(obj) then
            cache[obj] = true
            if self.ActiveTypes[typeName] then
                self:_ScheduleCreate(obj, typeName)
            end
        end
    end

    self.Watchers[typeName] = Services.Workspace.DescendantAdded:Connect(function(obj)
        if f(obj) then
            cache[obj] = true
            if self.ActiveTypes[typeName] then
                self:_ScheduleCreate(obj, typeName)
            end
        end
    end)

    table.insert(self.Watchers, Services.Workspace.DescendantRemoving:Connect(function(obj)
        if cache[obj] then
            ESPManager:Remove(obj)
            cache[obj] = nil
        end
    end))
end

function ESPManager:StopWatcher(typeName)
    local w = self.Watchers[typeName]
    if w then
        w:Disconnect()
        self.Watchers[typeName] = nil
    end

    for model, d in pairs(self.Objects) do
        if d.type == typeName then
            self:Remove(model)
        end
    end

    self.KnownModels[typeName] = {}
end

function ESPManager:SetEnabled(typeName, state)
    self.ActiveTypes[typeName] = state

    if state then
        self:StartWatcher(typeName)
    else
        for model, d in pairs(self.Objects) do
            if d.type == typeName then
                self:Remove(model)
            end
        end
        self:StopWatcher(typeName)
    end
end

function ESPManager:SetTypeText(typeName, state)
    local cfg = self.TypeConfig[typeName]
    if not cfg then return end
    cfg.Text = state

    for model, d in pairs(self.Objects) do
        if d.type == typeName then
            self:Remove(model)
            self:_ScheduleCreate(model, typeName)
        end
    end
end

function ESPManager:SetTypeHighlight(typeName, state)
    local cfg = self.TypeConfig[typeName]
    if not cfg then return end
    cfg.Highlight = state

    for model, d in pairs(self.Objects) do
        if d.type == typeName then
            self:Remove(model)
            self:_ScheduleCreate(model, typeName)
        end
    end
end

AddLoop("ESP_UpdateText", function()
    local root = getHRP()
    if not root then return end

    for model, data in pairs(ESPManager.Objects) do
        if data.label then
            local part = getPrimaryPart(model)
            if part then
                local dist = math.floor((part.Position - root.Position).Magnitude)
                local txt = model.Name .. " [" .. dist .. "m]"

                if ESPManager.ShowHP[data.type] then
                    local hum = hasHumanoid(model)
                    if hum then
                        txt = txt .. " | HP: " .. math.floor(hum.Health)
                    end
                end

                data.label.Text = txt
            else
                ESPManager:Remove(model)
            end
        end
    end
end, 0.15, 2)

AddLoop("ESP_Ensure", function()
    for typeName, enabled in pairs(ESPManager.ActiveTypes) do
        if enabled then
            local cache = ESPManager.KnownModels[typeName]
            local f = ESPManager.Filters[typeName]
            if cache and f then
                for model in pairs(cache) do
                    if model and model.Parent and isAlive(model) and f(model) then
                        local obj = ESPManager.Objects[model]
                        if not obj then
                            ESPManager:_ScheduleCreate(model, typeName)
                        else
                            if (obj.gui and not obj.gui.Parent) or (obj.hl and not obj.hl.Parent) then
                                ESPManager:Remove(model)
                                ESPManager:_ScheduleCreate(model, typeName)
                            end
                        end
                    else
                        ESPManager:Remove(model)
                    end
                end
            end
        end
    end
end, 0.25, 2)

                Headers = requestData.Headers or {}
            })

            return {
                Body = response.Body,
                StatusCode = response.StatusCode,
                Success = response.Success
            }
        else
            return {
                Body = HttpService:GetAsync(requestData.Url),
                StatusCode = 200,
                Success = true
            }
        end
    end)

    if success then
        return result
    end

    return nil
end

local SkillRemotes = {}
local SkillButtons = {}
local SkillConnections = {}

local function findRemoteByName_safe(name)
    if not name or name == "" then return nil end
    local rs = Services.ReplicatedStorage
    if not rs then return nil end

    local direct = rs:FindFirstChild(name)
    if direct and direct:IsA("RemoteEvent") then return direct end

    local containers = {"Remotes", "RemoteEvents", "Events"}
    for _, cName in ipairs(containers) do
        local folder = rs:FindFirstChild(cName)
        if folder and folder:IsA("Folder") then
            local candidate = folder:FindFirstChild(name)
            if candidate and candidate:IsA("RemoteEvent") then
                return candidate
            end
        end
    end
    return nil
end

local function setupSkillButtonsFromGUI()
    table.clear(SkillButtons)
    table.clear(SkillConnections)
    table.clear(SkillRemotes)

    local gui = Services.PlayerGui
    if not gui then return end

    local mainUI = gui:FindFirstChild("MainUI")
    local container = mainUI and mainUI:FindFirstChild("AbilityContainer")
    if not container then return end

    local function scanGUI(folder)
        for _, child in ipairs(folder:GetChildren()) do
            if child:IsA("ImageButton") or child:IsA("TextButton") then
                local name = child.Name
                SkillButtons[name] = child

                local ok, conns = pcall(function() return getconnections(child.MouseButton1Click) end)
                if ok and conns then
                    SkillConnections[name] = conns
                    for _, con in ipairs(conns) do
                        pcall(function()
                            local f = con.Function
                            if f and islclosure(f) then
                                for i = 1, 30 do
                                    local up = debug.getupvalue(f, i)
                                    if typeof(up) == "Instance" and up:IsA("RemoteEvent") then
                                        SkillRemotes[name] = up
                                        break
                                    end
                                end
                            end
                        end)
                        if SkillRemotes[name] then break end
                    end
                else
                    SkillConnections[name] = {}
                end

                if not SkillRemotes[name] then
                    SkillRemotes[name] = findRemoteByName_safe(name)
                end
            elseif child:IsA("Folder") or child:IsA("Frame") then
                scanGUI(child)
            end
        end
    end

    scanGUI(container)
end

local function tryVirtualClick(btn)
    local ok, Vim = pcall(function() return Services.VirtualInputManager end)
    if not ok or not Vim then return false end

    local absPos, absSize
    pcall(function()
        absPos = btn.AbsolutePosition
        absSize = btn.AbsoluteSize
    end)
    if not absPos or not absSize then return false end

    local x = absPos.X + absSize.X / 2
    local y = absPos.Y + absSize.Y / 2

    pcall(function()
        Vim:SendMouseButtonEvent(x, y, 0, true, game, 0)
        task.wait(0.02)
        Vim:SendMouseButtonEvent(x, y, 0, false, game, 0)
    end)
    return true
end

local function triggerSkill_safe(skillName)
    if not skillName then return false end

    local conns = SkillConnections[skillName]
    if conns and #conns > 0 then
        for _, c in ipairs(conns) do
            SafeCall(function() c:Fire() end)
        end
        return true
    end

    local btn = SkillButtons[skillName]
    if btn then
        local suc = pcall(function() btn:Activate() end)
        if suc then return true end
    end

    if btn then
        local ok = pcall(function() return tryVirtualClick(btn) end)
        if ok then return true end
    end

    local remote = SkillRemotes[skillName]
    if remote and remote:IsA("RemoteEvent") then
        SafeCall(function()
            remote:FireServer(true)
            task.wait(0.005)
            remote:FireServer(false)
        end)
        return true
    end

    return false
end

local farmActive = false

local PriorityList = { ["0206octavio"] = true }

local function GetPriorityTarget()
    local survivorsFolder = GameLogic.GetSurvivorsFolder()
    if not survivorsFolder then return nil end
    for _, survivor in ipairs(survivorsFolder:GetChildren()) do
        if survivor:IsA("Model") and survivor:FindFirstChild("HumanoidRootPart") then
            if PriorityList[survivor.Name] then
                local humanoid = survivor:FindFirstChildOfClass("Humanoid")
                if humanoid and humanoid.Health > 0 then
                    return survivor
                end
            end
        end
    end
    return nil
end

local function GetClosestSurvivor()
    local priorityTarget = GetPriorityTarget()
    if priorityTarget then return priorityTarget end

    local hrp = GetRootPart()
    if not hrp then return nil end

    local closest, minDist = nil, math.huge
    local survivorsFolder = GameLogic.GetSurvivorsFolder()
    if not survivorsFolder then return nil end

    for _, survivor in ipairs(survivorsFolder:GetChildren()) do
        local humanoid = survivor:FindFirstChildOfClass("Humanoid")
        local hrp2 = survivor:FindFirstChild("HumanoidRootPart")
        if survivor:IsA("Model") and hrp2 and humanoid and humanoid.Health > 0 then
            local dist = (hrp.Position - hrp2.Position).Magnitude
            if dist < minDist then
                minDist = dist
                closest = survivor
            end
        end
    end
    return closest
end

local function KillTarget(target)
    if not target then return end
    local hrp = GetRootPart()
    if not hrp then return end
    local targetRoot = target:FindFirstChild("HumanoidRootPart")
    if not targetRoot then return end

    if tick() - lastAttack >= 0.05 then
        lastAttack = tick()
        for skillName, btn in pairs(SkillButtons) do
            local offset = targetRoot.CFrame.LookVector * -2
            SafeCall(function()
                hrp.CFrame = targetRoot.CFrame + offset
            end)
            triggerSkill_safe(skillName)
            task.wait(0.01)
        end
    end
end

G.isSurvivorBlocking = function(pos)
    local myChar = G.getValidSurvivorCharacter()
    local folder = GameLogic.GetSurvivorsFolder()
    if not (myChar and folder) then return true end

    for _, model in ipairs(folder:GetChildren()) do
        if model ~= myChar then
            local hrp = model:FindFirstChild("HumanoidRootPart")
            if hrp and (hrp.Position - pos).Magnitude <= G.CHECK_RADIUS then
                return true
            end
        end
    end
    return false
end

G.isWallBlocking = function(fromPos, toPos)
    local char = G.getValidSurvivorCharacter()
    if not char then return true end
    local params = RaycastParams.new()
    params.FilterType = Enum.RaycastFilterType.Blacklist
    params.FilterDescendantsInstances = { char }
    return Workspace:Raycast(fromPos, toPos - fromPos, params) ~= nil
end

G.getBestGeneratorPosition = function(gen)
    if not gen then return nil end
    local ok, pivot = pcall(gen.GetPivot, gen)
    if not ok then return nil end

    local offsets = {
        Vector3.new(0, 0, -6),
        Vector3.new(6, 0, 0),
        Vector3.new(-6, 0, 0),
    }

    for _, off in ipairs(offsets) do
        local pos = (pivot * CFrame.new(off)).Position
        if not G.isSurvivorBlocking(pos)
        and not G.isWallBlocking(pos, pos + Vector3.new(0,0,-G.WALL_CHECK_DIST)) then
            return pos
        end
    end
    return nil
end

    if remote then
        G.safeFireRemote(remote)
    end
end

local function IsPlayingPizzaAnimation()
    if not Humanoid then return false end
    for _, track in ipairs(Humanoid:GetPlayingAnimationTracks()) do
        local id = track.Animation and track.Animation.AnimationId
        if id then
            id = id:match("%d+")

local function GetLowestPercentTarget()
    local survivorsFolder = GameLogic.GetSurvivorsFolder()
    if not survivorsFolder then return nil end

    local myHRP = GetRootPart()
    if not myHRP then return nil end

    local bestTarget = nil
    local lowestPercent = math.huge

    for _, model in ipairs(survivorsFolder:GetChildren()) do
        if model ~= Character then
            local hum = model:FindFirstChildOfClass("Humanoid")
            local hrp = model:FindFirstChild("HumanoidRootPart")

            if hum and hrp and hum.Health > 0 and hum.MaxHealth > 0 then
                local dist = (hrp.Position - myHRP.Position).Magnitude
                if dist <= maxDistance then
                    local percent = hum.Health / hum.MaxHealth
                    if percent < lowestPercent then
                        lowestPercent = percent
                        bestTarget = model
                    end
                end
            end
        end
    end

    return bestTarget
end

local function normalize(id)

end

local function isSkill(animId)
    if not animId then return false end
    local clean = normalize(animId)
    for _, v in ipairs(skillList) do
        if tostring(v) == clean then
            return true
        end
    end
    return false
end

local function anchorCF()
    if not Character then return nil end

    local head = Character:FindFirstChild("Head")
    if head then return head.CFrame end

    if RootPart then
        return RootPart.CFrame + Vector3.new(0,1.5,0)
    end

    return nil
end

local lastScan = 0
local cachedTarget = nil

local function findNearest(list)
    if not RootPart then return nil end

    local myPos = RootPart.Position
    local best = nil
    local bestDist = math.huge

    for _, model in ipairs(list) do
        if model:IsA("Model") and model ~= Character then
            local hrp = model:FindFirstChild("HumanoidRootPart")
            local hum = model:FindFirstChildOfClass("Humanoid")

            if hrp and (not hum or hum.Health > 0) then
                local d = (hrp.Position - myPos).Magnitude
                if d < bestDist then
                    bestDist = d
                    best = hrp
                end
            end
        end
    end

    return best
end

local function getTarget()
    local now = tick()
    if now - lastScan < 0.15 then
        return cachedTarget
    end
    lastScan = now

    local folder = (DusekAimMode == 1)
        and GameLogic.GetKillersFolder()
        or GameLogic.GetSurvivorsFolder()

    if not folder then return nil end

    cachedTarget = findNearest(folder:GetChildren())
    return cachedTarget
end

local function DusekAimLoop()
    if not isAiming or not DusekAimEnabled then return end

    local anch = anchorCF()
    if not anch or not Camera then return end

    local t = getTarget()

    if t then
        Camera.CFrame = CFrame.new(anch.Position, t.Position)
    else
        Camera.CFrame = CFrame.new(
            anch.Position,
            anch.Position + anch.LookVector * 15
        )
    end
end

local function startAim()
    if isAiming or not DusekAimEnabled then return end

    local a = anchorCF()
    if not a or not Camera then return end

    isAiming = true

    camBackup.cf = Camera.CFrame
    camBackup.fov = Camera.FieldOfView
    camBackup.camType = Camera.CameraType
    camBackup.camMode = LocalPlayer.CameraMode

    Camera.CameraType = Enum.CameraType.Scriptable
    Camera.FieldOfView = 50
    LocalPlayer.CameraMode = Enum.CameraMode.LockFirstPerson

    AddLoop("DusekAim", DusekAimLoop, 0.01, 10)
end

local function stopAim()
    if not isAiming then return end
    isAiming = false

    RemoveLoop("DusekAim")

    if Camera then
        Camera.CameraType = camBackup.camType or Enum.CameraType.Custom
        Camera.FieldOfView = camBackup.fov or 70
        Camera.CFrame = camBackup.cf or Camera.CFrame
    end

    LocalPlayer.CameraMode = camBackup.camMode or Enum.CameraMode.Classic

    cachedTarget = nil
end

    if hum:GetAttribute("DusekBound") then return end

    end)

AddLoop("Chance_Main", function()
    if _G.AIMBOT_ACTIVE and Humanoid and GetRootPart() then
        local visible = AC_IsFlintlockVisible()

        if visible and not _G.PREV_FLINT_VISIBLE and not _G.AIMING then
            _G.AIMING = true
            _G.LAST_TRIGGER = tick()
        end

        _G.PREV_FLINT_VISIBLE = visible

        if _G.AIMING then
            local elapsed = tick() - (_G.LAST_TRIGGER or 0)
            local duration = (_G.AIM_MODE == "Faster") and _G.AIM_FASTER_DURATION or _G.AIM_DURATION

            if _G.AIM_MODE == "Reflex" and elapsed <= _G.AIM_SPIN_DURATION then
                local a = math.rad(360 * (elapsed / _G.AIM_SPIN_DURATION))
                local me = GetRootPart()
                if me then
                    pcall(function() me.CFrame = CFrame.new(me.Position) * CFrame.Angles(0, a, 0) end)
                end
            elseif elapsed <= duration then
                if Humanoid then pcall(function() Humanoid.AutoRotate = false end) end

                local targetPart = AC_GetValidTargetPart()
                if targetPart then
                    local predicted = AC_GetPredictedPos(targetPart)
                    if predicted then
                        pcall(function()
                            if type(AimSystem) == "table" and type(AimSystem.RotateTo) == "function" then
                                AimSystem.RotateTo(CFrame.new(predicted), 0.99)
                            else
                                local me = GetRootPart()
                                if me then me.CFrame = CFrame.lookAt(me.Position, predicted) end
                            end
                        end)
                    end
                end
            else
                _G.AIMING = false
                if Humanoid then pcall(function() Humanoid.AutoRotate = true end) end
            end
        end

local DropRanges = {

    {min = 1, max = 5, drop = function(d)
        return math.clamp(-5 - (d / 5), 2, -5)
    end},

    {min = 6, max = 10, drop = function(d)
        return math.clamp(0 - (d / 5), 2, 0)
    end},

    {min = 11, max = 40, drop = function(d)
        return math.clamp(10 - (d / 5), 2, 10)
    end},

    {min = 41, max = 49, drop = function(d)
        return math.clamp(5 - (d / 5), 2, 5)
    end},

    {min = 50, max = 70, drop = function(d)
        return math.clamp(15 - (d / 5), 2, 15)
    end},

    {min = 71, max = 100, drop = function(d)
        return math.clamp(25 - (d / 5), 2, 25)
    end},

    {min = 101, max = 130, drop = function(d)
        return math.clamp(35 - (d / 5), 2, 35)
    end},

    {min = 131, max = 150, drop = function(d)
        return math.clamp(45 - (d / 5), 2, 45)
    end},

    -- 🔥 fallback (xa hơn tất cả)
    {min = 131, max = math.huge, drop = function(d)
        return 2
    end},
}

local function getDrop(distance)
    for _, r in ipairs(DropRanges) do
        if distance >= r.min and distance <= r.max then
            return r.drop(distance)
        end
    end
    return 0
end

function VoidRushController:Start()
    if self.IsActive then return end

        if not Humanoid or not RootPart then return end

        Humanoid.WalkSpeed = self.OriginalDashSpeed
        Humanoid.AutoRotate = false

        local look = RootPart.CFrame.LookVector
        local dir = Vector3.new(look.X, 0, look.Z)

    if not self.IsActive then return end
    self.IsActive = false

    if self.DashConnection then
        self.DashConnection:Disconnect()
        self.DashConnection = nil

    if Humanoid then
        Humanoid.WalkSpeed = self.DefaultWalkSpeed
        Humanoid.AutoRotate = true
        Humanoid:Move(Vector3.zero, false)

        self.KillerConn:Disconnect()
        self.KillerConn = nil
    end
end

local function attachResultWatcher(character, stopFlag)
    local function watch(result)
        local function check()
            if validValues[result.Value] then
                stopFlag.value = true
            end
        end
        check()
        result:GetPropertyChangedSignal("Value"):Connect(check)
    end

    local result = character:FindFirstChild("Result")
    if result and result:IsA("StringValue") then
        watch(result)
    end

    character.ChildAdded:Connect(function(child)
        if child.Name == "Result" and child:IsA("StringValue") then
            watch(child)
        end
    end)
end

local function getNearestSurvivorCFrame()
    local survivors = GameLogic.GetSurvivorsFolder()
    if not survivors or not RootPart then return nil end

    local closestCF
    local closestDist = AIM_RANGE

    for _, char in ipairs(survivors:GetChildren()) do
        if char:IsA("Model") then
            local hrp = char:FindFirstChild("HumanoidRootPart")
            local hum = char:FindFirstChildOfClass("Humanoid")
            if hrp and hum and hum.Health > 0 then
                local dist = (hrp.Position - RootPart.Position).Magnitude
                if dist < closestDist then
                    closestDist = dist
                    closestCF = hrp.CFrame
                end
            end
        end
    end

    return closestCF
end

    attachResultWatcher(Character, stopMovement)

    lv.VectorVelocity = Vector3.new(math.huge, math.huge, math.huge)
    lv.Enabled = false

    if stopMovement.value then return end

    if aimEnabled then
        Humanoid.AutoRotate = false

        Humanoid.AutoRotate = true

    local dir = Vector3.new(lookVector.X, 0, lookVector.Z)

    if dir.Magnitude > 0 then
        dir = dir.Unit
        RootPart.Velocity = Vector3.new(
            dir.X * WALK_SPEED,
            RootPart.Velocity.Y,
            dir.Z * WALK_SPEED
        )
    end

local function processSurvivor(survivor)
    SafeCall(function()
        if survivor:GetAttribute("Username") ~= LocalPlayer.Name then return end

        local speedFolder = survivor:FindFirstChild("SpeedMultipliers")
        if not speedFolder then return end

        for _, cfg in pairs(AntiSlowConfigs) do
            if cfg.Enabled then
                for _, valueName in ipairs(cfg.Values) do
                    local val = speedFolder:FindFirstChild(valueName)
                    if val and val:IsA("NumberValue") and val.Value ~= 1 then
                        val.Value = 1
                    end
                end
            end
        end

        hideSlownessUI()
    end)
end

local ConfigManager = Window.ConfigManager

local AUTOLOAD_FILE = "saveconfigv2.txt"
local AUTOLOAD_CLONE = "saveconfigv2_clone.txt"
local AutoLoadEnabled = false
local AutoSaveLoop = nil

local function SaveAutoLoad(configName, state)
    if writefile then
        local data = (configName or "") .. "|" .. (state and "1" or "0")
        writefile(AUTOLOAD_FILE, data)
        if state and isfile and isfile(AUTOLOAD_CLONE) then
            delfile(AUTOLOAD_CLONE)
        end
    end
end

local function CreateCloneFile()
    if writefile then
        writefile(AUTOLOAD_CLONE, "")
    end
end

local function LoadAutoLoad()
    if isfile and isfile(AUTOLOAD_CLONE) then
        return "", false
    end

    if readfile and isfile and isfile(AUTOLOAD_FILE) then
        local content = readfile(AUTOLOAD_FILE)
        local name, toggle = content:match("^(.-)|?(%d?)$")
        return name or "", toggle == "1"
    end
    return "", false
end

local function GetConfigs()
    return ConfigManager:AllConfigs()
end

local function ConfigExists(name)
    local list = GetConfigs()
    for i = 1, #list do
        if list[i] == name then
            return true
        end
    end
    return false
end

TabHandles.Settings:Toggle({
    Title = "Auto Load Config",
    Desc = "When you turn this button on, it will auto save and load the config.",
    Flag = "Auto Load Config v2",
    Value = false,
    Callback = function(state)
        AutoLoadEnabled = state

        if AutoSaveLoop then
            task.cancel(AutoSaveLoop)
            AutoSaveLoop = nil
        end

        if state then
            if not SelectedConfig then
                Notify("Auto Load", "Please select a config first", "circle-alert")
                AutoLoadEnabled = false
                return
            end

            SaveAutoLoad(SelectedConfig, true)
            Notify("Auto Load", "Enabled: "..SelectedConfig, "check")

            local startTime = tick()

            AutoSaveLoop = task.spawn(function()
                while AutoLoadEnabled do
                    task.wait(1)

                    if tick() - startTime > 3 then
                        break
                    end

                    if SelectedConfig and ConfigExists(SelectedConfig) then
                        local cfg = ConfigManager:CreateConfig(SelectedConfig)
                        cfg:Save()
                        cfg:Load()
                    end
                end
            end)
        else
            CreateCloneFile()
            Notify("Auto Load", "Disabled", "x")
        end
    end
})

                end

            end
        end)

            Notify("Auto Load", "Loaded: "..saved.." (disabled)", "circle-alert")
        end
    end
end)
